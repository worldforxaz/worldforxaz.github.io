    把分数坐标批量改成整数坐标

    import os
import numpy as np
for i in range(452):  # 从0到100
    filename = f"aa.{i}"  # 构建输入文件名
    output_filename = f"{i}.vasp"  # 构建输出文件名

    if os.path.exists(filename):
        # 读取POSCAR文件
        with open(filename, "r") as f:
            poscar = f.read().splitlines()

        # 从POSCAR文件中提取晶格矢量
        lattice_vectors = np.array([list(map(float, line.split())) for line in poscar[2:5]])

        # 从POSCAR文件中提取分数坐标
        frac_coords = np.array([list(map(float, line.split())) for line in poscar[8:]])

        # 将分数坐标转换为整数坐标
        cart_coords = np.dot(frac_coords, lattice_vectors)

        # 替换分数坐标为整数坐标
        for j in range(len(cart_coords)):
            poscar[8 + j] = " ".join(map(str, cart_coords[j]))

        # 保存转换后的POSCAR文件
        with open(output_filename, "w") as f:
            f.write("\n".join(poscar))
    else:
        print(f"File {filename} does not exist.")

print("Conversion completed.")

    import argparse
import numpy as np
import pandas as pd
import string
parser = argparse.ArgumentParser(description='处理XDATCAR，得到某个原子所有步x坐标值和前后两步的差')
parser.add_argument('-nst','--nstatoms',type=int,help='第多少个原子')
parser.add_argument('path',help='文件路径')
args=parser.parse_args()
def sub(x,y):
    '''
    求差
    '''
    subtraction = y-x
    return subtraction
def readXD(path):
    '''
    读入XDATCAR
    '''
    with open(path) as f:
        line_after7=f.readlines()[6:]
    natoms = sum([int(x) for x in line_after7[0].strip(' ').strip('\n').split(' ') if x != ''])
    nstep=(int(((len(line_after7)-1)/(natoms+1))))
    line_all=[]
    i=2
    for j in np.linspace(1,100,nstep):
        line_all.append(line_after7[i:i+natoms])
        i=i+natoms+1
    '''
    变为numpy数组
    '''
    j=0
    struc_list=[]
    for j in range(nstep):
        A=np.zeros((natoms,3))
        A_row=0
        for line in line_all[j]:
            list=line.strip(' ').strip('\n').split(' ')
            list=[x for x in list if x !='']
            float_list=[float(x) for x in list]
            A[A_row:]=float_list[0:3]
            A_row +=1
        struc_list.append(A)
    struc_all = np.array(struc_list)
    return struc_all,natoms,nstep
def export_x(path,nst):
    '''
    输出第n个原子x的位置
    '''
    struc_list,natoms,nstep= readXD(path)
    st_x=[struc_list[i][nst-1][0] for i in range(nstep)]
    st=pd.DataFrame(st_x)
    st.to_csv('position.csv',index=False,header=False)
    return
def subtraction(path,nst):
    '''
    求相邻位置x的差
    '''
    struc_list,natoms,nstep= readXD(path)
    sublist=[]
    for i in range(1,nstep):
        subtraction = sub(struc_list[i][nst-1][0],struc_list[i-1][nst-1][0])
        sublist.append(subtraction)
    subst = pd.DataFrame(sublist)
    subst.to_csv('sub.csv',index=False,header=False,float_format='%.10f')
    return

    import math  
import numpy
from pymatgen.core import Structure
from pymatgen.io.vasp.inputs import Poscar

def transform_coordinates(x1, y1,x2,y2, angle):
    x=x2-x1
    y=y2-y1
#    a=math.sqrt(x**2 + y**2)
    # 计算x'
    x_prime = x*math.cos(angle) - y*math.sin(angle) 
    # 计算y'  
    y_prime = x*math.sin(angle) + y*math.cos(angle)  
    aa = math.sqrt(x_prime**2 + y_prime**2)
    return x_prime, y_prime

poscar = Poscar.from_file('POSCAR')
structure = poscar.structure
#设置旋转点
rotation_point = structure[0]
x_coord,y_coord,_= rotation_point.coords
#设置旋转角度/设置每步旋转的角度
ten = math.pi/18
for a in range(1,37):
    m = ten
#设置要旋转的原子的次序
    for atom in structure[1:]:
        x_coord_rotation, y_coord_rotation, _ = atom.coords
        new_x,new_y= transform_coordinates(x_coord,y_coord,x_coord_rotation,y_coord_rotation,m)
        new_x1 = new_x + x_coord
        new_y1 = new_y + y_coord
#注意下面有一个递增的效果，因为是直接修改的原来的坐标
        atom.coords = [new_x1,new_y1,atom.coords[2]]
    new_poscar = Poscar(structure)
    new_poscar.write_file(f"{a}.vasp")
